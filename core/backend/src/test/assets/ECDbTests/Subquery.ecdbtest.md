# CTE subquery

- dataset: AllProperties.bim

```sql
WITH
  VersionAliases AS (
    SELECT
      VersionMajor,
      GROUP_CONCAT(Alias) AS Aliases
    FROM
      ECDbMeta :ECSchemaDef
    GROUP BY
      VersionMajor
  )
SELECT
  te.ECInstanceId,
  va.Aliases
FROM
  aps.TestElement te
  LEFT JOIN VersionAliases va ON te.i + 100 = va.VersionMajor;
```

# Nested Subquery with HAVING clause

- dataset: AllProperties.bim

```sql
SELECT
  VersionMajor,
  COUNT(*) AS VersionMajorCount
FROM
  meta.ECSchemaDef s
GROUP BY
  VersionMajor
HAVING
  COUNT(*) > (
    SELECT
      AVG(VersionMinorCount)
    FROM
      (
        SELECT
          VersionMinor,
          COUNT(*) AS VersionMinorCount
        FROM
          meta.ECSchemaDef
        GROUP BY
          VersionMinor
      ) sub
  );
```

```sql
        SELECT
          AVG(VersionMajor)

          ECDbMeta :ECSchemaDef

              Alias

              ECDbMeta :ECSchemaDef
```
